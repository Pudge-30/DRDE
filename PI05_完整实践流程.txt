cat > evaluate_pi05_inference.py << 'EOF'
"""使用 π0.5 在 LIBERO 环境中推理"""
import torch
from lerobot.envs.factory import make_env
from lerobot.policies.pi05 import PI05Policy
from lerobot.policies.factory import make_pre_post_processors

# 配置
device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
model_id = "lerobot/pi05_libero_finetuned"

print(f"Device: {device}")
print(f"Loading model: {model_id}")

# 1. 加载模型
policy = PI05Policy.from_pretrained(model_id)
policy.to(device)
policy.eval()

# 2. 创建预处理器
preprocess, postprocess = make_pre_post_processors(
    policy.config,
    model_id
)

# 3. 创建环境
print("\nCreating LIBERO environment...")
env = make_env(
    env_type="libero",
    task="libero_spatial",
    n_envs=1,
)

# 4. 运行评估
n_episodes = 3
n_success = 0

for episode in range(n_episodes):
    print(f"\n{'='*60}")
    print(f"Episode {episode + 1}/{n_episodes}")

    obs, info = env.reset()
    done = False
    step = 0
    max_steps = 300

    while not done and step < max_steps:
        # 构建输入
        obs_dict = {
            "observation.state": torch.from_numpy(obs["agent_pos"]).float().unsqueeze(0).to(device),
            "observation.images.image": torch.from_numpy(obs["pixels"]["image"]).float().unsqueeze(0).to(device),
            "observation.images.image2": torch.from_numpy(obs["pixels"]["image2"]).float().unsqueeze(0).to(device),
        }

        # 推理
        obs_processed = preprocess(obs_dict)
        with torch.no_grad():
            action = policy.select_action(obs_processed)
        action = postprocess(action)

        # 执行
        obs, reward, terminated, truncated, info = env.step(action.cpu().numpy().squeeze(0))
        done = terminated or truncated
        step += 1

    success = info.get("is_success", False)
    n_success += int(success)
    print(f"  Steps: {step}")
    print(f"  Success: {success}")

print(f"\n{'='*60}")
print(f"Success Rate: {n_success}/{n_episodes} = {100*n_success/n_episodes:.1f}%")
print("✓ Evaluation completed!")
EOF

4.3 用 Python 自定义训练脚本（可选）

cat > train_pi05_custom.py << 'EOF'
"""自定义 π0.5 训练脚本"""
from pathlib import Path
import torch
from torch.utils.data import DataLoader
from accelerate import Accelerator
from lerobot.datasets import LeRobotDataset, LeRobotDatasetMetadata
from lerobot.policies.pi05 import PI05Policy
from lerobot.policies.factory import make_pre_post_processors

# 配置
output_dir = Path("outputs/pi05_custom")
output_dir.mkdir(parents=True, exist_ok=True)

# 1. 初始化 Accelerator
accelerator = Accelerator(
    mixed_precision="bf16",
    gradient_accumulation_steps=1,
)

print(f"Device: {accelerator.device}")
print(f"Process: {accelerator.process_index}/{accelerator.num_processes}")

# 2. 加载数据集
print("\nLoading LIBERO dataset...")
dataset = LeRobotDataset("HuggingFaceVLA/libero")
metadata = LeRobotDatasetMetadata("HuggingFaceVLA/libero")
print(f"Dataset size: {len(dataset)} frames")

# 3. 加载预训练模型
print("\nLoading π0.5 model...")
policy = PI05Policy.from_pretrained("lerobot/pi05_base")
policy.train()

# 4. 创建处理器
preprocessor, _ = make_pre_post_processors(policy.config, metadata.stats)

# 5. 优化器
optimizer = torch.optim.AdamW(
    policy.parameters(),
    lr=1e-5,
    weight_decay=1e-4
)

# 6. DataLoader
dataloader = DataLoader(
    dataset,
    batch_size=8,
    shuffle=True,
    num_workers=4,
    pin_memory=True,
    drop_last=True
)

# 7. 准备分布式训练
policy, optimizer, dataloader = accelerator.prepare(
    policy, optimizer, dataloader
)

# 8. 训练循环
print("\nStarting training...")
total_steps = 1000

for step, batch in enumerate(dataloader):
    if step >= total_steps:
        break

    # 预处理
    batch = preprocessor(batch)

    # 前向传播
    with accelerator.autocast():
        loss, output_dict = policy.forward(batch)

    # 反向传播
    accelerator.backward(loss)

    # 梯度裁剪
    if accelerator.sync_gradients:
        accelerator.clip_grad_norm_(policy.parameters(), 1.0)

    # 优化器步骤
    optimizer.step()
    optimizer.zero_grad()

    # 日志
    if step % 100 == 0:
        print(f"Step {step}/{total_steps} | Loss: {loss.item():.4f}")

    # 保存检查点
    if step % 500 == 0 and step > 0:
        accelerator.wait_for_everyone()
        if accelerator.is_main_process:
            unwrapped_model = accelerator.unwrap_model(policy)
            checkpoint_dir = output_dir / f"checkpoint-{step}"
            unwrapped_model.save_pretrained(checkpoint_dir)
            print(f"✓ Checkpoint saved: {checkpoint_dir}")

# 9. 保存最终模型
accelerator.wait_for_everyone()
if accelerator.is_main_process:
    unwrapped_model = accelerator.unwrap_model(policy)
    final_dir = output_dir / "final_model"
    unwrapped_model.save_pretrained(final_dir)
    print(f"\n✓ Training completed! Model saved: {final_dir}")
EOF

# 运行训练
python train_pi05_custom.py

================================================================================
步骤 6：故障排查
================================================================================

常见问题 1：MuJoCo 渲染错误
────────────────────────────────────────────────────────────────────────────
错误信息：
  ERROR: GLEW initialization error: Missing GL version

解决方案：
  export MUJOCO_GL=osmesa
  # 或者
  export MUJOCO_GL=egl


常见问题 2：显存不足
────────────────────────────────────────────────────────────────────────────
错误信息：
  RuntimeError: CUDA out of memory

解决方案：
  1. 减小 batch_size：--batch_size=8 或 --batch_size=4
  2. 确保使用梯度检查点：--policy.gradient_checkpointing=true
  3. 使用混合精度：--policy.dtype=bfloat16


常见问题 3：数据集下载缓慢
────────────────────────────────────────────────────────────────────────────
解决方案：
  # 设置 HuggingFace 镜像（中国大陆）
  export HF_ENDPOINT=https://hf-mirror.com

  # 或者手动下载数据集后指定本地路径
  # （具体方法见 LeRobot 文档）


常见问题 4：Transformers 版本不兼容
────────────────────────────────────────────────────────────────────────────
错误信息：
  ImportError: cannot import name 'XXX' from 'transformers'

解决方案：
  # π0.5 需要特定的 Transformers 分支
  pip uninstall transformers -y
  pip install -e ".[pi]"


常见问题 5：评估时环境卡住不动
────────────────────────────────────────────────────────────────────────────
解决方案：
  1. 检查 MuJoCo 渲染后端：export MUJOCO_GL=egl
  2. 检查是否有足够的系统资源
  3. 尝试减少并行环境数：--eval.batch_size=1

注意事项
================================================================================

1. 显存管理
   - π0.5 是 3B 参数模型，需要较大显存
   - 建议使用 24GB+ VRAM 的 GPU
   - 必须启用 gradient_checkpointing

2. 训练时间
   - 快速测试（100步）：~5-10分钟
   - 完整训练（6000步）：~8-16小时（取决于GPU）

3. 数据集下载
   - LIBERO 数据集约 370KB（元数据）+ 数GB（视频）
   - 首次加载会自动下载到 ~/.cache/huggingface/

4. 评估建议
   - 先用少量剧集（n_episodes=2）测试
   - 确认流程正常后再运行完整评估（n_episodes=10）

5. W&B 使用
   - 建议启用 wandb 记录训练过程
   - 需要先登录：wandb login
